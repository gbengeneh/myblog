SELECT 
    p.id, 
    p.title, 
    p.content, 
    p.image, 
    p.category_id, 
    p.user_id, 
    p.created_at, 
    p.updated_at, 
    c.name AS category_name, 
    u.username
FROM posts p
LEFT JOIN categories c ON p.category_id = c.id
LEFT JOIN users u ON p.user_id = u.id
WHERE p.id = ?
LIMIT 1;
